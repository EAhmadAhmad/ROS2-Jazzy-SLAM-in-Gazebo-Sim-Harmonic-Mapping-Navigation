% Problem Statement and Goals
\begin{frame}{Problem Statement and Goals}
    \textbf{Problem:}
    \begin{itemize}
        \item Autonomous robots need to understand unknown environments
        \item SLAM is fundamental for robot navigation
        \item Need practical implementation with real robots
    \end{itemize}
    
    \textbf{Project Goals:}
    \begin{enumerate}
        \item Create a realistic house environment in Gazebo
        \item Implement Cartographer SLAM with TurtleBot3
        \item Build accurate map of the environment
        \item Enable navigation within the mapped space
        \item Apply software engineering principles
    \end{enumerate}
\end{frame}

% Solution Description
\begin{frame}{Solution Description}
    \textbf{Methods and Algorithms:}
    \begin{itemize}
        \item \textbf{SLAM Algorithm:} Google Cartographer
        \begin{itemize}
            \item Graph-based SLAM with loop closure
            \item Real-time pose graph optimization
            \item 2D occupancy grid mapping
        \end{itemize}
        \item \textbf{Localization:} Adaptive Monte Carlo Localization (AMCL)
        \item \textbf{Path Planning:} Nav2 with Dijkstra's algorithm
    \end{itemize}
    
    \textbf{Tools and Frameworks:}
    \begin{itemize}
        \item ROS2 Jazzy (Robotics middleware)
        \item Gazebo Harmonic (Physics simulation)
        \item TurtleBot3 (Robot platform)
        \item Docker (Containerization)
    \end{itemize}
\end{frame}
